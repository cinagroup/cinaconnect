📝 Usage Example

      
<!-- Step 1: Configure -->
<script>
  window.CinaConnect = {
    projectId: 'YOUR_PROJECT_ID',
    theme: 'dark',
    chains: [1, 10, 137]
  };
</script>

<!-- Step 2: Load with SRI -->
<script
  src="https://cdn.cinacoin.dev/v0.2.0/cinacoin.js"
  integrity="sha384-XXXXX"
  crossorigin="anonymous"
></script>

<!-- Step 3: Render -->
<script>
  CinaConnect.renderConnectButton('#connect-button', {
    size: 'lg',
    onConnect: (addr) => console.log('Connected:', addr)
  });
</script>
