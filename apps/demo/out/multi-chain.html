unified-api.ts
1// CinaConnect — Unified Multi-Chain API
2// One interface. Every chain. Zero complexity.
3
4import { CinaConnect } from '@cinaconnect/sdk';
5
6const client = new CinaConnect();
7
8// 🔗 Connect to ANY chain with the same API
9const eth = await client.connect('ethereum', 'MetaMask');
10const sol = await client.connect('solana', 'Phantom');
11const btc = await client.connect('bitcoin', 'Xverse');
12const ton = await client.connect('ton', 'Tonkeeper');
13
14// 💰 Read balances across chains
15const balances = await Promise.all([
16  client.getBalance('ethereum', eth.address),
17  client.getBalance('solana', sol.address),
18  client.getBalance('bitcoin', btc.address),
19  client.getBalance('ton', ton.address),
20]);
21
22// ⚡ Cross-chain transfer
23const tx = await client.transfer({
24  from: { chain: 'ethereum', address: eth.address },
25  to:   { chain: 'solana',   address: sol.address },
26  amount: '0.5 ETH',
27  slippage: 0.5,
28});
29
30// 📡 Listen to events on all chains
31client.on('transaction', (event) => {
32  console.log(`[${event.chain}] ${event.type}: ${event.hash}`);
33});
